// ฟังก์ชันสำหรับดึงข้อมูลลมแบบ GFS และสร้างแอนิเมชัน function fetchWindData() { // เปลี่ยนลิงก์เป็นข้อมูลลมแบบครอบคลุมทั่วโลก (Global Wind Data ตัวอย่าง) const windDataUrl = 'https://raw.githubusercontent.com/danwild/leaflet-velocity/master/demo/wind-global.json'; fetch(windDataUrl) .then(res => res.json()) .then(data => { const velocityLayer = L.velocityLayer({ displayValues: true, displayOptions: { velocityType: 'ลมผิวพื้น', displayPosition: 'bottomleft', displayEmptyString: 'ไม่มีข้อมูลทิศทางลมในบริเวณนี้' }, data: data, maxVelocity: 15, velocityScale: 0.01, // ปรับค่านี้ให้เป็น 0.01 เพื่อให้เส้นลมพัดยาวขึ้นและมองเห็นง่ายขึ้น particleAge: 90, lineWidth: 2, colorScale: ["#a1a1aa", "#64748b", "#475569", "#334155", "#1e293b"] // โทนสีเทา-ดำ }); // นำเส้นลมไปใส่ใน Layer ที่เตรียมไว้ velocityLayer.addTo(windLayerGroup); }) .catch(err => { console.error("เกิดข้อผิดพลาดในการโหลดข้อมูลทิศทางลม", err); showToast("ไม่สามารถโหลดข้อมูลทิศทางลมได้ในขณะนี้", "error"); }); }
